                function toggle_gif(img, icon){
                
                    if($(img).data('state') == 0){
                        play_gif(img, icon);
                    } else {
                        stop_gif(img, icon);
                    }
                }

                function play_gif(img, icon){
                    $(img).attr('src', $(img).data('animation'));
                    $(img).data('state', 1);
                    $(icon).fadeOut();
                }

                function stop_gif(img, icon){
                    $(img).attr('src', $(img).data('original'));
                    $(img).data('state', 0);
                    $(icon).fadeIn();
                }
